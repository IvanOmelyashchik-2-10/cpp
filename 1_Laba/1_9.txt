#include <iostream>
using namespace std;
int main()
{
	unsigned int a = 1, b = 0, c = 1, b1 = 0, b2 = 0; //где б1 первая операция(а^б), б2 вторая ((а^б)+с)
	b1 = a ^ b; // 00000001 ^ 00000000 результат 00000001
	b2 = b1 | c; // 00000001 + 000000001
	cout << "First:" << b2 << endl; // равно 1 в десятичной
	a = 0, b = 1, c = 0;
	b1 = a ^ b; // 00000000 ^ 00000001 результат 00000001
	b2 = b1 | c; //00000001 + 00000000 результат 00000001
	cout<< "Second:" << b2 << endl; // результат 1 в десятичной
	a = 1, b = 2, c = 1;
	b1 = a ^ b; // 00000001 ^ 00000010 результат 00000011
	b2 = b1 | c; // 00000011 + 00000001 результат 00000011
	cout<<"Third:" << b2 << endl; // 3 в десятичной
	return 0;
}
